import matplotlib.pyplot as plt

characters = list(letter_freq.keys())
frequencies = list(letter_freq.values())

plt.barh(range(len(characters)), frequencies, tick_label=characters)
plt.xlabel('Frequency')
plt.ylabel('Characters')
plt.title('Character Frequency Distribution')

plt.show()

Question 2¶

  1. Generate a sequence of at least 5 random numbers (integers are fine), and visualize the data using both scatter and plot on different axes. You may use the index of the value as its x value in the plot.
In [17]:
import numpy as np
import matplotlib.pyplot as plt

# 5 random integers
np.random.seed(42)  # Set a seed for reproducibility
random_numbers = np.random.randint(1, 100, size=5)

# Two subplots
fig, (ax1, ax2) = plt.subplots(nrows=2, ncols=1, figsize=(8, 6))

# Scatter plot
ax1.scatter(range(len(random_numbers)), random_numbers, c='red', label='Data Points')
ax1.set_xlabel('Index')
ax1.set_ylabel('Values')
ax1.set_title('Scatter Plot')
ax1.legend()

# Line plot
ax2.plot(range(len(random_numbers)), random_numbers, marker='o', linestyle='-', color='blue', label='Line Plot')
ax2.set_xlabel('Index')
ax2.set_ylabel('Values')
ax2.set_title('Line Plot')
ax2.legend()

plt.tight_layout()
plt.show()
